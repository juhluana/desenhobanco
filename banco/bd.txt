-- Criar banco de dados
CREATE DATABASE desenhos_animados CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci;
USE desenhos_animados;

-- Criar tabela de desenhos animados
CREATE TABLE catalogo (
  id INT AUTO_INCREMENT PRIMARY KEY,
  titulo VARCHAR(100) NOT NULL,
  genero VARCHAR(50) NOT NULL,
  ano INT CHECK (ano >= 1900),
  estudio VARCHAR(100),
  tipo ENUM('Série Animada', 'Filme Animado') DEFAULT 'Série Animada',
  avaliacao DECIMAL(3,1) DEFAULT 0.0
);
